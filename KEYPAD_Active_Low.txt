//active low input
#include <stdio.h>
#include <mega16.h>
#include <delay.h>
void main() {
	DDRB = 0xFF;  // Set PORTB as output (SEgments)
    	DDRD = 0x00;  // Set PORTD as input (Push Buttons)
    while(1) 
	    {
		int mode = PIND;
		switch (mode) {
			case 0x77 : //P0
				PORTB= 0xBF;//0
			break;
			case 0x7B : //P1
				PORTB= 0x86;//1
			break;
			case 0x7D : //P2
				PORTB= 0xD7;//2
			break;
			case 0x7E : //P3
				PORTB= 0x8F;//3
			break;
			case 0xB7 : //P4
				PORTB= 0xE6;//4
			break;
			case 0xBB : //P5
				PORTB= 0xED;//5
			break;
			case 0xBD : //P6
				PORTB= 0xFD;//6
			break;
			case 0xBE : //P7
				PORTB= 0x87;//7
			break;
			case 0xD7 : //P8
				PORTB= 0xFF;//8
			break;
			case 0xDB : //P9
				PORTB= 0xEF;//9
			break;
			case 0xDD : //P10
				PORTB= 0xF7;//A
			break;
			case 0xDE : //P11
				PORTB= 0xFC;//B
			break;
			case 0xE7 : //P12
				PORTB= 0xD8;//C
			break;
			case 0xEB : //P13
				PORTB= 0xDE;//D
			break;
			case 0xED : //P14
				PORTB= 0xF9;//E
			break;
			case 0xEE : //P15
				PORTB= 0xF1;//F
			break;
		}
	}	
}
