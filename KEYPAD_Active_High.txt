#include <stdio.h>
#include <mega16.h>
#include <delay.h>
void main() {
	DDRB = 0xFF;  // Set PORTB as output (SEgments)
    	DDRD = 0x00;  // Set PORTD as input (Push Buttons)
    while(1) 
	    {
		int mode = PIND;
		switch (mode) {
			case 0x88 : //P0
				PORTB= 0xBF;//0
			break;
			case 0x84 : //P1
				PORTB= 0x86;//1
			break;
			case 0x82 : //P2
				PORTB= 0xD7;//2
			break;
			case 0x81 : //P3
				PORTB= 0x8F;//3
			break;
			case 0x48 : //P4
				PORTB= 0xE6;//4
			break;
			case 0x44 : //P5
				PORTB= 0xED;//5
			break;
			case 0x42 : //P6
				PORTB= 0xFD;//6
			break;
			case 0x41 : //P7
				PORTB= 0x87;//7
			break;
			case 0x28 : //P8
				PORTB= 0xFF;//8
			break;
			case 0x24 : //P9
				PORTB= 0xEF;//9
			break;
			case 0x22 : //P10
				PORTB= 0xF7;//A
			break;
			case 0x21 : //P11
				PORTB= 0xFC;//B
			break;
			case 0x18 : //P12
				PORTB= 0xD8;//C
			break;
			case 0x14 : //P13
				PORTB= 0xDE;//D
			break;
			case 0x12 : //P14
				PORTB= 0xF9;//E
			break;
			case 0x11 : //P15
				PORTB= 0xF1;//F
			break;
		}
	}	
}
